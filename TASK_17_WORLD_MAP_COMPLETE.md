### Using the WorldMap Component

```tsx

const regions = [
  {
    id: 'region-1',
    name: 'My Region',
    description: 'A cool place',
    lore: 'Ancient quote',
    status: 'unlocked',
    levelRequirement: 10,
    difficulty: 2,
    completionPercentage: 50,
    x: 200,
    y: 300,
    connectsTo: ['region-2'],
    challenges: [
      {
        id: 'challenge-1',
        name: 'Boss Fight',
        type: 'boss',
        difficulty: 2,
        completed: false,
        reward: '5000 XP',
      },
    ],
    rewards: ['Title', '10000 Gold'],
    bonus: '+10% XP',
  },
];

<WorldMap
  regions={regions}
  currentLevel={47}
  onRegionClick={(region) => console.log(region)}
  onFastTravel={(regionId) => console.log('Travel to', regionId)}

/>
```
